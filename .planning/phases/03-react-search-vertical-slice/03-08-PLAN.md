    CONFIG_RAW=$(curl -s -w '\n%{http_code}' "$CONFIG_ENDPOINT" -H "X-Correlation-Id: $SMOKE_CORR_ID")
    CONFIG_BODY=$(printf '%s' "$CONFIG_RAW" | sed '$d')
    CONFIG_CODE=$(printf '%s' "$CONFIG_RAW" | tail -n 1)
    if [[ "$CONFIG_CODE" != "200" ]]; then
      echo "FAIL: /config returned HTTP $CONFIG_CODE"
      exit 1
    fi

    # Assert fileName category exists
    if ! printf '%s' "$CONFIG_BODY" | grep -q '"key"[[:space:]]*:[[:space:]]*"fileName"'; then
      echo "FAIL: /config response missing fileName category"
      exit 1

    # grep across the whole response is sufficient — search-config-v4.json only
    # references each renderer key once per category.
    for KEY in appIDCellRenderer supportEmailCellRenderer executionOrderButtonRenderer; do

        echo "FAIL: /config response missing cellRenderer \"$KEY\" — Phase 3 React registry expects this string key"
        exit 1
      fi
    done

    echo "Step 0 PASS — /config exposes fileName with the 3 Phase 3 renderer keys"
